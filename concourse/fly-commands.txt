a few examples using concourse's fly command

HOST=
PORT=
USERNAME=
PASSWORD=

# login to a running concourse
fly login \
    --target stemmarest \
    --concourse-url http://${HOST}:${PORT}
    --username ${USERNAME}
    --password ${PASSWORD}

# set a pipeline, will with existing on
fly set-pipeline \
    --target stemmarest \
    --pipeline stemmarest \
    --config pipeline.yml \
    # load variables from file
    --load-vars-from credentials.yml \
    # give a single variables
    --var "<SOME_BOOLEAN>=true"
    --var "<RATHER_COMPLEX_VALUES>=$(cat <SOME_FILE>)"

fly unpause-pipeline \
    --target stemmarest \
    --pipeline stemmarest

# live output of running jobs. especially if there's a lot
# of output this works better than in browser
fly watch \
    --target stemmarest \
    --job <pipeline>/<job>

# run a single task
fly execute \
    --target stemmarest
    --config tasks/test.yml
    --inputs-from=stemmarest/job-test

# log into a previous container to look around
fly hijack \
    --target stemmarest \
    --job <pipeline>/<job> \
    <command>
